pub fn values<V>(self, values: V) -> Selfwhere V: Into<Row<'a>>,

Adds a new row to be inserted.

let query = Insert::multi_into("users", vec!["foo"])
    .values(vec![1])
    .values(vec![2]);

let (sql, params) = Sqlite::build(query)?;

assert_eq!("INSERT INTO `users` (`foo`) VALUES (?), (?)", sql);

assert_eq!(
    vec![
        Value::from(1),
        Value::from(2),
    ], params);
